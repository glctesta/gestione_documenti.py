Private Sub Badge_AfterUpdate()
If Len(Nz(Me.Badge.Value, "")) = 0 Then Exit Sub
    On Error GoTo Errori
    Dim manager() As String
    Dim i As Integer

    Set newReferat = New clsReferat
    
    manager = Split(VerificaBadge2(Me.Badge.Value, , True), ",")
    With newReferat
        For i = LBound(manager) To UBound(manager)
            If manager(i) = "NO QUALIFY" Then
                MsgBox "You are not qualify to issue a disciplinary claim.", vbExclamation + vbOKOnly, Intestazione()
                Me.Detail.Visible = False
                Exit Sub
            End If
            
             If Len(Nz(manager(0), "")) = 0 Then
                MsgBox "Your badge cannot be used to issue this kind of document.", vbExclamation + vbOKOnly, Intestazione()
                    GoTo Fine
             End If
            
            
            Select Case i
                Case Is = 0
                    .ManagerId = manager(i)
                Case Is = 1
                    .ManagerName = manager(i)
                Case Is = 2
                    .CostCenterId = manager(i)
                Case Is = 3
                    .CostCenterName = manager(i)
                Case Is = 4
                    .IsSuperUser = IIf(CByte(manager(i)) = 9, -1, 0)
                Case Is = 8
                    .GlobalAccess = IIf(CByte(manager(i)) >= 90, -1, 0)
            End Select
        
        Next i
        
    
    Me.Badge.Value = ""
    Me.Testata.Value = "Department " & Me.Department.Value & ", procedura pornita de catre <FONT COLOR= BLUE><B>" & .ManagerName & "</FONT></B>"
    Me.Detail.Visible = True
    End With
    
     Call CaricaComboDipendenti(newReferat.CostCenterId, newReferat.GlobalAccess)
     Call CaricaMotivi
     Me.EmployeeBadge.SetFocus
     Me.EmployeeBadge.Dropdown
     
     GoTo Fine

Errori:
  MsgBox Err.Number & " " & Err.Description & vbCrLf & "Form: " & Form.Name, vbCritical + vbOKOnly, Intestazione()

Fine:
End Sub

Private Sub DataAvvenimento_AfterUpdate()
    If Me.DataAvvenimento.Value > Date Then
        MsgBox "Data referat e gresita!", vbExclamation + vbOKOnly, Intestazione()
        Me.DataAvvenimento.Value = Empty
        Me.Badge.SetFocus
        Me.DataAvvenimento.SetFocus
        Exit Sub
    ElseIf DateDiff("D", Me.DataAvvenimento.Value, Date) > 2 Then
        MsgBox "Cannot record an evend older that 3 days.", vbExclamation + vbOKOnly, Intestazione()
        Me.DataAvvenimento.Value = Empty
        Me.Badge.SetFocus
        Me.DataAvvenimento.SetFocus
        Exit Sub
    End If
    newReferat.DataAvvenimento = Me.DataAvvenimento.Value
End Sub

Private Function VerificaEssenziali() As Boolean
    Dim Crt As control
    Dim Valido As Boolean
 
        Valido = True
        
        For Each Crt In Me.Detail.Controls
            If Crt.Tag = "Essenziali" Then
                If Len(Nz(Crt.Value, "")) = 0 Then
                    Valido = False
                    Crt.BackColor = vbRed
                Else
                    If Crt.Name = "MotivoReferat" Then
                        Crt.BackColor = RGB(231, 230, 230)
                    Else
                        Crt.BackColor = vbWhite
                    End If
                End If
            End If
        Next
        
        VerificaEssenziali = Valido
        
End Function

Private Function HaAltriReferati(EmployeeHireHistoryId As Long, Optional QUaleTipo As Byte = 0) As Byte
    Dim cnn As ADODB.Connection
    Dim Rcs As ADODB.Recordset

    Set Rcs = New ADODB.Recordset
    Set cnn = New ADODB.Connection
    cnn.ConnectionString = GetCnnString(CBool(DLookup("[lWork]", "TbSys", "[dataout] Is Null")))
    cnn.ConnectionTimeout = 60
    cnn.Open

    If QUaleTipo = 0 Then
        Set Rcs = cnn.Execute("select count(  k.EmployeeHireHistoryId) AS NoReferate,YEAR(K.EventDate) REFERATEyEAR from (select DISTINCT e.EmployeeName + ' ' + e.EmployeeSurname AS EmployeeName, r.DocName AS DisciplinaryDocumentNo,d.DataAvvenimento as EventDate,d.ExplicationNote, r.RegistroId, h.EmployeeHireHistoryId from employee.[dbo].[EmployeeDisciplinaryHistory] AS D inner join employee.dbo.employeehirehistory as H on h.EmployeeHireHistoryId=d.EmployeeHireHistoryId inner join employee.dbo.employees as E on e.EmployeeId=H.EmployeeId inner join employee.dbo.registry AS R on r.RegistroId=D.RegistroId inner join employee.dbo.EmployeeHireHistoryDocs AS DH on r.RegistroId=dh.RegistroId and right(dh.PathAndDoc,3)='PDF' WHERE H.EmployeeHireHistoryId=" & EmployeeHireHistoryId & ") as K GROUP BY YEAR(K.EventDate) order by  YEAR(K.EventDate) DESC;")
        If Rcs.EOF Then
            HaAltriReferati = 0
        Else
            HaAltriReferati = Rcs.Fields(0)
        End If
    Else
        Dim Storia As String
        Storia = Me.EmployeeBadge.Column(2) & " are " & HaAltriReferati(EmployeeHireHistoryId, 0) & " referat in total, si sunt:" & vbCrLf
        Set Rcs = cnn.Execute("SELECT  ROW_NUMBER() OVER ( ORDER BY K.RegistroId) as NoReferat   ,YEAR(K.EventDate) As AnReferate ,cast(k.DisciplinaryDocumentNo as text) as fd,cast(k.ExplicationNote as text) as jha FROM (SELECT DISTINCT e.EmployeeName + ' ' + e.EmployeeSurname AS EmployeeName ,r.DocName AS DisciplinaryDocumentNo ,d.DataAvvenimento AS EventDate     ,d.ExplicationNote      ,r.RegistroId       ,h.EmployeeHireHistoryId FROM employee.[dbo].[EmployeeDisciplinaryHistory] AS D  INNER JOIN employee.dbo.employeehirehistory AS H        ON h.EmployeeHireHistoryId = d.EmployeeHireHistoryId    INNER JOIN employee.dbo.employees AS E      ON e.EmployeeId = H.EmployeeId  INNER JOIN employee.dbo.registry AS R       ON r.RegistroId = D.RegistroId  INNER JOIN employee.dbo.EmployeeHireHistoryDocs AS DH       ON r.RegistroId = dh.RegistroId         AND RIGHT(dh.PathAndDoc, 3) = 'PDF'     WHERE H.EmployeeHireHistoryId = " & EmployeeHireHistoryId & "     ) AS K ORDER BY YEAR(K.EventDate) DESC;")
        
        Do Until Rcs.EOF
            Storia = Storia & Rcs.Fields(0) & ") " & Rcs.Fields(1) & " " & Rcs.Fields(3) & vbCrLf
            Rcs.MoveNext
        Loop
        
        MsgBox Storia, vbOKOnly + vbInformation, Intestazione()
    
    End If


    GoTo Fine



Fine:
    cnn.Close
    Set cnn = Nothing
    Set Rcs = Nothing
    
End Function
